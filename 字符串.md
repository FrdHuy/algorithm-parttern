题解思路：[[labuladong]如何实现一个计算器]( https://leetcode.cn/problems/basic-calculator-iii/solutions/2099856/labuladong-ru-he-shi-xian-yi-ge-ji-suan-84mpt/ )
```java
class Solution {  
    public int calculate(String s) {  
        Deque<Character> queue = new LinkedList<>();  
        for (char c : s.toCharArray()) {  
            queue.add(c);  
        }  
        return helper(queue);  
    }  
  
    public int helper(Deque<Character> queue) {  
        int num = 0;  
        char sign = '+';  
        Stack<Integer> stack = new Stack<>();  
  
        while (!queue.isEmpty()) {  
            char c = queue.poll();  
            // 如果c是数字，则计算num  
            if (Character.isDigit(c)) {  
                num = num * 10 + (c - '0');  
            }  
  
            if (c == '(') {  
                num = helper(queue);  
            }  
  
            // 如果c不是数字或者是最后一位（说明遇到了下一个符号），则把之前的数字存入stack  
            // 并且在最后重置num和sign  
            // 注意空格可能会影响到结果，在if statement中加入判断  
            if ((!Character.isDigit(c) && c != ' ') || queue.isEmpty()) {  
                int prenum;  
                switch (sign) {  
                    // 加号情况  
                    case '+':  
                        stack.push(num);  
                        break;                    // 减号情况  
                    case '-':  
                        stack.push(-num);  
                        break;                    // 乘号情况  
                    case '*':  
                        prenum = stack.pop();  
                        stack.push(prenum * num);  
                        break;                    // 除号情况  
                    case '/':  
                        prenum = stack.pop();  
                        stack.push(prenum / num);  
                        break;                }  
                sign = c;  
                num = 0;  
            }  
              
            if (c == ')') {  
                break;  
            }  
        }  
        // 求和  
        int res = 0;  
        while (!stack.isEmpty()) {  
            res += stack.pop();  
        }  
        return res;  
    }  
}
